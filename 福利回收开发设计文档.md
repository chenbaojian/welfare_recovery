### 1.2 技术选型

| 层级 | 技术栈 | 说明 |
|------|--------|------|
| 前端 | 微信小程序原生 + Vant Weapp | 原生开发保证性能，Vant组件库提升开发效率 |
| 后端 | Node.js + Express/Koa 或 Java + Spring Boot | 根据团队技术栈选择 |
| 数据库 | MySQL 8.0 | 主数据存储，支持事务 |
| 缓存 | Redis 6.0 | 会话管理、数据缓存、分布式锁 |
| 消息队列 | RabbitMQ / RocketMQ | 异步处理、订单状态通知 |
| 文件存储 | 阿里云OSS / 腾讯云COS | 图片、文件存储 |
| 网关 | Nginx | 反向代理、负载均衡、SSL |

### 3.3 中间件设计

#### 3.3.1 认证中间件

```javascript
const authMiddleware = async (ctx, next) => {
  const token = ctx.headers['authorization'];
  
  if (!token) {
    ctx.throw(401, '未登录');
  }
  
  try {
    // 验证token
    const payload = jwt.verify(token, config.jwt.secret);
    
    // 检查token是否在黑名单
    const isBlacklisted = await redis.get(`token:blacklist:${token}`);
    if (isBlacklisted) {
      ctx.throw(401, '登录已过期');
    }
    
    // 挂载用户信息
    ctx.state.user = payload;
    
    await next();
```

### 4.2 网络请求封装

```javascript
// utils/request.js
const BASE_URL = 'https://api.example.com';
const TOKEN_KEY = 'token';

class Request {
  constructor() {
    this.baseURL = BASE_URL;
    this.timeout = 10000;
  }

  /**
   * 请求拦截器
   */
  interceptRequest(options) {
    const token = wx.getStorageSync(TOKEN_KEY);
    
    return {
      ...options,
      header: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : '',
        ...options.header
      }
    };
  }

  /**
   * 响应拦截器
   */
  interceptResponse(response) {
    const { statusCode, data } = response;
    
    if (statusCode === 401) {
      // token过期，跳转登录
      wx.removeStorageSync(TOKEN_KEY);
      wx.redirectTo({ url: '/pages/login/login' });
      return Promise.reject(new Error('登录已过期'));
    }
    
    if (statusCode === 403) {
      // 未实名认证
      wx.navigateTo({ url: '/pages/verify/verify' });
      return Promise.reject(new Error('请先完成实名认证'));
    }
    
    if (data.code !== 200) {
      return Promise.reject(new Error(data.message || '请求失败'));
    }
    
    return data.data;
  }

  /**
   * 通用请求方法
   */
  request(options) {
    const config = this.interceptRequest(options);
    
    return new Promise((resolve, reject) => {
      wx.request({
        ...config,
        url: this.baseURL + config.url,
        timeout: this.timeout,
        success: (res) => {
          this.interceptResponse(res)
            .then(resolve)
            .catch(reject);
        },
        fail: (err) => {
          reject(new Error('网络异常，请检查网络后重试'));
        }
      });
    });
  }

  get(url, data, options = {}) {
    return this.request({ ...options, url, data, method: 'GET' });
  }

  post(url, data, options = {}) {
    return this.request({ ...options, url, data, method: 'POST' });
  }
}

export default new Request();
```

```javascript
const CryptoJS = require('crypto-js');

const AES_KEY = process.env.AES_KEY;  // 32位密钥
const AES_IV = process.env.AES_IV;    // 16位向量

/**
 * AES加密
 */
function encrypt(data) {
  const key = CryptoJS.enc.Utf8.parse(AES_KEY);
  const iv = CryptoJS.enc.Utf8.parse(AES_IV);
  
  const encrypted = CryptoJS.AES.encrypt(
    CryptoJS.enc.Utf8.parse(data),
    key,
    {
      iv: iv,
      mode: CryptoJS.mode.CBC,
      padding: CryptoJS.pad.Pkcs7
    }
  );
  
  return encrypted.toString();
}

/**
 * AES解密
 */
function decrypt(encryptedData) {
  const key = CryptoJS.enc.Utf8.parse(AES_KEY);
  const iv = CryptoJS.enc.Utf8.parse(AES_IV);
  
  const decrypted = CryptoJS.AES.decrypt(
    encryptedData,
    key,
    {
      iv: iv,
      mode: CryptoJS.mode.CBC,
      padding: CryptoJS.pad.Pkcs7
    }
  );
  
  return decrypted.toString(CryptoJS.enc.Utf8);
}

/**
 * 手机号脱敏
 */
function maskPhone(phone) {
  if (!phone || phone.length !== 11) return phone;
```

### 5.2 Token管理

```javascript
// utils/auth.js
const jwt = require('jsonwebtoken');
const redis = require('../config/redis');

const JWT_SECRET = process.env.JWT_SECRET;
const JWT_EXPIRES_IN = '7d';
const TOKEN_PREFIX = 'token:';

/**
 * 生成Token
 */
async function generateToken(userId) {
  const payload = {
    userId,
    timestamp: Date.now()
  };
  
  const token = jwt.sign(payload, JWT_SECRET, {
    expiresIn: JWT_EXPIRES_IN
  });
  
  // 存储到Redis
  await redis.set(
    `${TOKEN_PREFIX}${userId}`,
    token,
    'EX',
    7 * 24 * 60 * 60
  );
  
  return token;
}

/**
 * 验证Token
 */
async function verifyToken(token) {
  try {
    const payload = jwt.verify(token, JWT_SECRET);
    
    // 检查是否在黑名单
    const isBlacklisted = await redis.get(`token:blacklist:${token}`);
    if (isBlacklisted) {
      throw new Error('Token已失效');
    }
    
    // 检查是否与Redis中的一致
    const storedToken = await redis.get(`${TOKEN_PREFIX}${payload.userId}`);
    if (storedToken !== token) {
      throw new Error('Token不匹配');
    }
    
    return payload;
  } catch (err) {
    throw new Error('Token无效');
  }
}

/**
 * 使Token失效
 */
async function invalidateToken(userId, token) {
  // 加入黑名单
  await redis.set(
```

### 5.3 接口签名验证

```javascript
// middleware/signature.js
const crypto = require('crypto');

const SIGNATURE_KEY = process.env.SIGNATURE_KEY;

/**
 * 生成签名
 */
function generateSign(params, timestamp) {
  // 1. 参数排序
  const sortedKeys = Object.keys(params).sort();
  
  // 2. 拼接字符串
  const str = sortedKeys
    .map(key => `${key}=${params[key]}`)
    .join('&') + `&timestamp=${timestamp}&key=${SIGNATURE_KEY}`;
  
  // 3. MD5加密
  return crypto.createHash('md5').update(str).digest('hex');
}

/**
 * 签名验证中间件
 */
const signatureMiddleware = (ctx, next) => {
  const { sign, timestamp, ...params } = ctx.request.body;
  
  // 检查时间戳（5分钟内有效）
  const now = Date.now();
  if (Math.abs(now - timestamp) > 5 * 60 * 1000) {
    ctx.throw(400, '请求已过期');
  }
  
  // 验证签名
  const expectedSign = generateSign(params, timestamp);
  if (sign !== expectedSign) {
    ctx.throw(400, '签名验证失败');
  }
  
  return next();
};

module.exports = signatureMiddleware;
```

```javascript
// utils/cache.js
const CACHE_PREFIX = 'cache:';
const DEFAULT_EXPIRE = 5 * 60 * 1000; // 5分钟

/**
 * 设置缓存
 */
function set(key, data, expire = DEFAULT_EXPIRE) {
  const cacheData = {
    data,
    expire: Date.now() + expire
  };
  
  wx.setStorageSync(CACHE_PREFIX + key, cacheData);
}

/**
 * 获取缓存
 */
function get(key) {
  const cacheData = wx.getStorageSync(CACHE_PREFIX + key);
  
  if (!cacheData) return null;
  
  // 检查是否过期
  if (Date.now() > cacheData.expire) {
    wx.removeStorageSync(CACHE_PREFIX + key);
    return null;
  }
  
  return cacheData.data;
}

/**
 * 清除缓存
 */
function clear(pattern) {
  const res = wx.getStorageInfoSync();
  res.keys.forEach(key => {
    if (key.startsWith(CACHE_PREFIX) && 
        (!pattern || key.includes(pattern))) {
      wx.removeStorageSync(key);
    }
  });
}

module.exports = { set, get, clear };
```
